def create_bmi_card(self, parent, bmi):
    """р╕кр╕гр╣Йр╕▓р╕З Card р╣Бр╕кр╕Фр╕З BMI"""

    # р╕Бр╕│р╕лр╕Щр╕Фр╕кр╕╡р╕Хр╕▓р╕б BMI

        color = "#E74C3C"  # р╣Бр╕Фр╕З
        status = "р╕Хр╣Ир╕│р╕Бр╕зр╣Ир╕▓р╣Ар╕Бр╕Ур╕Ср╣М"
        icon = "ЁЯФ┤"

        status = "р╕Ыр╕Бр╕Хр╕┤"

def calculate_grade(self, score):
    """р╕Др╕│р╕Щр╕зр╕Ур╣Ар╕Бр╕гр╕Фр╕Ир╕▓р╕Бр╕Др╕░р╣Бр╕Щр╕Щ"""

        grade = 4.0
        color = "#27AE60"  # р╣Ар╕Вр╕╡р╕вр╕зр╣Ар╕Вр╣Йр╕б
    elif score >= 75:
        grade = 3.5
        color = "#52BE80"  # р╣Ар╕Вр╕╡р╕вр╕зр╕нр╣Ир╕нр╕Щ
    elif score >= 70:
        grade = 3.0
        color = "#52BE80"
    elif score >= 65:
        grade = 2.5
        color = "#F1C40F"  # р╣Ар╕лр╕ер╕╖р╕нр╕З
    elif score >= 60:
        grade = 2.0
        color = "#F1C40F"
    elif score >= 55:
        grade = 1.5
        color = "#E67E22"  # р╕кр╣Йр╕б
    elif score >= 50:
        grade = 1.0
        color = "#E67E22"
    else:
        grade = 0.0
        color = "#E74C3C"  # р╣Бр╕Фр╕З

    return grade, color

def show_grade_realtime(self):
    """р╣Бр╕кр╕Фр╕Зр╣Ар╕Бр╕гр╕Ф realtime р╣Ар╕бр╕╖р╣Ир╕нр╕Юр╕┤р╕бр╕Юр╣Мр╕Др╕░р╣Бр╕Щр╕Щ"""

    try:
        score = float(self.score_var.get())
        grade, color = self.calculate_grade(score)

        # р╣Бр╕кр╕Фр╕З Card
        self.grade_card.configure(fg_color=color)
        self.grade_label.configure(text=f"{grade:.1f}")
    except:
        self.grade_label.configure(text="-")
